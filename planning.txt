-- Getting a snapshot to copy the exercises from the routine to the workout
INSERT INTO workouts_exercises (workout_id, exercise_id, exercise_index)
SELECT
    :workout_id,
    exercise_id,
    exercise_index
FROM routines_exercises
WHERE routine_id = :routine_id


-- Creating the sets per exercise
INSERT INTO workouts_sets (workout_id, exercise_id, set_index, notes)
SELECT
    :workout_id,
    re.exercise_id,
    gs.set_index,
    re.exercise_notes
FROM routines_exercises re
CROSS JOIN generate_series(1, re.planned_sets) gs(set_index)
WHERE re.routine_id = :routine_id



Endpoints :

1. POST /workouts - Creates a workout (routine_id in the body)
2. GET /workouts - Get all workouts (start_date, end_date as query params)
3. GET /workouts/{workout_id} - Get a workout
4. PATCH /workouts/{workout_id} - Update a workout (begda or endda can be send in the body)
